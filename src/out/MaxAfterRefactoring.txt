private action _MilageRecordTable_ max ()      
returns (      
  max : Number)      
do       
  max := _MilageRecordTable table_ -> row at(0) -> _MileageRecordColumn_      
  for each ct in _MilageRecordTable_ table      
    where true      
  do       
    if ct -> _MileageRecordColumn_ > max then       
      max := ct -> _MileageRecordColumn_       
    else do nothing
